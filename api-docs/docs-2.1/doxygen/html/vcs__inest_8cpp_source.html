297  /* ***************************************** */

303  }

306  /* **************************************************** */

308  if (xl < 0.0) {

311  /* *************************************************** */

313  } else {

317  /* *************************************************** */

319  } else {
320  /* *************************************************** */
321  /* **** ACCEPT RESULTS OF FORCER ********************* */
322  /* *************************************************** */
323  par = par * 2.0 * xl;
324  }
325  }
