186 {

209 void StFlow::eval(size_t jg, doublereal* xg,
210  doublereal* rg, integer* diagg, doublereal rdt)
211 {
212  // if evaluating a Jacobian, and the global point is outside the domain of
213  // influence for this domain, then skip evaluating the residual
214  if (jg != npos && (jg + 1 < firstPoint() || jg > lastPoint() + 1)) {
215  return;
216  }
217 
218  // if evaluating a Jacobian, compute the steady-state residual
219  if (jg != npos) {
220  rdt = 0.0;
221  }
222 
223  // start of local part of global arrays
224  doublereal* x = xg + loc();
225  doublereal* rsd = rg + loc();
226  integer* diag = diagg + loc();
227 
228  size_t jmin, jmax;
229  if (jg == npos) { // evaluate all points
230  jmin = 0;
231  jmax = m_points - 1;
232  } else { // evaluate points for Jacobian
233  size_t jpt = (jg == 0) ? 0 : jg - firstPoint();
234  jmin = std::max<size_t>(jpt, 1) - 1;
235  jmax = std::min(jpt+1,m_points-1);
236  }
237 
238  updateProperties(jg, x, jmin, jmax);
239  evalResidual(x, rsd, diag, rdt, jmin, jmax);
240 }

242 void StFlow::updateProperties(size_t jg, double* x, size_t jmin, size_t jmax)
243 {
244  // properties are computed for grid points from j0 to j1
245  size_t j0 = std::max<size_t>(jmin, 1) - 1;
246  size_t j1 = std::min(jmax+1,m_points-1);
247 
248  updateThermo(x, j0, j1);
249  if (jg == npos || m_force_full_update) {
250  // update transport properties only if a Jacobian is not being
251  // evaluated, or if specifically requested
252  updateTransport(x, j0, j1);
253  }
254  if (jg == npos) {
255  double* Yleft = x + index(c_offset_Y, jmin);
256  m_kExcessLeft = distance(Yleft, max_element(Yleft, Yleft + m_nsp));
257  double* Yright = x + index(c_offset_Y, jmax);
258  m_kExcessRight = distance(Yright, max_element(Yright, Yright + m_nsp));
259  }
260 
261  // update the species diffusive mass fluxes whether or not a
262  // Jacobian is being evaluated
263  updateDiffFluxes(x, j0, j1);
264 }

266 void StFlow::evalResidual(double* x, double* rsd, int* diag,
267  double rdt, size_t jmin, size_t jmax)
268 {
269  //----------------------------------------------------
270  // evaluate the residual equations at all required
271  // grid points
272  //----------------------------------------------------
273 
274  // calculation of qdotRadiation
275 
276  // The simple radiation model used was established by Y. Liu and B. Rogg [Y.
277  // Liu and B. Rogg, Modelling of thermally radiating diffusion flames with
278  // detailed chemistry and transport, EUROTHERM Seminars, 17:114-127, 1991].
279  // This model uses the optically thin limit and the gray-gas approximation
280  // to simply calculate a volume specified heat flux out of the Planck
281  // absorption coefficients, the boundary emissivities and the temperature.
282  // The model considers only CO2 and H2O as radiating species. Polynomial
283  // lines calculate the species Planck coefficients for H2O and CO2. The data
284  // for the lines is taken from the RADCAL program [Grosshandler, W. L.,
285  // RADCAL: A Narrow-Band Model for Radiation Calculations in a Combustion
286  // Environment, NIST technical note 1402, 1993]. The coefficients for the
287  // polynomials are taken from [http://www.sandia.gov/TNF/radiation.html].
288 
289  if (m_do_radiation) {
290  // variable definitions for the Planck absorption coefficient and the
291  // radiation calculation:
292  doublereal k_P_ref = 1.0*OneAtm;
293 
294  // polynomial coefficients:
295  const doublereal c_H2O[6] = {-0.23093, -1.12390, 9.41530, -2.99880,
296  0.51382, -1.86840e-5};
297  const doublereal c_CO2[6] = {18.741, -121.310, 273.500, -194.050,
298  56.310, -5.8169};
299 
300  // calculation of the two boundary values
301  double boundary_Rad_left = m_epsilon_left * StefanBoltz * pow(T(x, 0), 4);
302  double boundary_Rad_right = m_epsilon_right * StefanBoltz * pow(T(x, m_points - 1), 4);
303 
304  // loop over all grid points
305  for (size_t j = jmin; j < jmax; j++) {
306  // helping variable for the calculation
307  double radiative_heat_loss = 0;
308 
309  // calculation of the mean Planck absorption coefficient
310  double k_P = 0;
311  // absorption coefficient for H2O
312  if (m_kRadiating[1] != npos) {
313  double k_P_H2O = 0;
314  for (size_t n = 0; n <= 5; n++) {
315  k_P_H2O += c_H2O[n] * pow(1000 / T(x, j), (double) n);
316  }
317  k_P_H2O /= k_P_ref;
318  k_P += m_press * X(x, m_kRadiating[1], j) * k_P_H2O;
319  }
320  // absorption coefficient for CO2
321  if (m_kRadiating[0] != npos) {
322  double k_P_CO2 = 0;
323  for (size_t n = 0; n <= 5; n++) {
324  k_P_CO2 += c_CO2[n] * pow(1000 / T(x, j), (double) n);
325  }
326  k_P_CO2 /= k_P_ref;
327  k_P += m_press * X(x, m_kRadiating[0], j) * k_P_CO2;
328  }
329 
330  // calculation of the radiative heat loss term
331  radiative_heat_loss = 2 * k_P *(2 * StefanBoltz * pow(T(x, j), 4)
332  - boundary_Rad_left - boundary_Rad_right);
333 
334  // set the radiative heat loss vector
335  m_qdotRadiation[j] = radiative_heat_loss;
336  }
337  }
338 
339  for (size_t j = jmin; j <= jmax; j++) {
340  //----------------------------------------------
341  // left boundary
342  //----------------------------------------------
343 
344  if (j == 0) {
345  // these may be modified by a boundary object
346 
347  // Continuity. This propagates information right-to-left, since
348  // rho_u at point 0 is dependent on rho_u at point 1, but not on
349  // mdot from the inlet.
350  rsd[index(c_offset_U,0)] =
351  -(rho_u(x,1) - rho_u(x,0))/m_dz[0]
352  -(density(1)*V(x,1) + density(0)*V(x,0));
353 
354  // the inlet (or other) object connected to this one will modify
355  // these equations by subtracting its values for V, T, and mdot. As
356  // a result, these residual equations will force the solution
357  // variables to the values for the boundary object
358  rsd[index(c_offset_V,0)] = V(x,0);
359  if (doEnergy(0)) {
360  rsd[index(c_offset_T,0)] = T(x,0);
361  } else {
362  rsd[index(c_offset_T,0)] = T(x,0) - T_fixed(0);
363  }
364  rsd[index(c_offset_L,0)] = -rho_u(x,0);
365 
366  // The default boundary condition for species is zero flux. However,
367  // the boundary object may modify this.
368  double sum = 0.0;
369  for (size_t k = 0; k < m_nsp; k++) {
370  sum += Y(x,k,0);
371  rsd[index(c_offset_Y + k, 0)] =
372  -(m_flux(k,0) + rho_u(x,0)* Y(x,k,0));
373  }
374  rsd[index(c_offset_Y + leftExcessSpecies(), 0)] = 1.0 - sum;
375 
376  // set residual of poisson's equ to zero
377  rsd[index(c_offset_P, 0)] = x[index(c_offset_P, j)];
378  } else if (j == m_points - 1) {
379  evalRightBoundary(x, rsd, diag, rdt);
380  // set residual of poisson's equ to zero
381  rsd[index(c_offset_P, j)] = x[index(c_offset_P, j)];
382  } else { // interior points
383  evalContinuity(j, x, rsd, diag, rdt);
384  // set residual of poisson's equ to zero
385  rsd[index(c_offset_P, j)] = x[index(c_offset_P, j)];
386 
387  //------------------------------------------------
388  // Radial momentum equation
389  //
390  // \rho dV/dt + \rho u dV/dz + \rho V^2
391  // = d(\mu dV/dz)/dz - lambda
392  //-------------------------------------------------
393  rsd[index(c_offset_V,j)]
394  = (shear(x,j) - lambda(x,j) - rho_u(x,j)*dVdz(x,j)
395  - m_rho[j]*V(x,j)*V(x,j))/m_rho[j]
396  - rdt*(V(x,j) - V_prev(j));
397  diag[index(c_offset_V, j)] = 1;
398 
399  //-------------------------------------------------
400  // Species equations
401  //
402  // \rho dY_k/dt + \rho u dY_k/dz + dJ_k/dz
403  // = M_k\omega_k
404  //-------------------------------------------------
405  getWdot(x,j);
406  for (size_t k = 0; k < m_nsp; k++) {
407  double convec = rho_u(x,j)*dYdz(x,k,j);
408  double diffus = 2.0*(m_flux(k,j) - m_flux(k,j-1))
409  / (z(j+1) - z(j-1));
410  rsd[index(c_offset_Y + k, j)]
411  = (m_wt[k]*(wdot(k,j))
412  - convec - diffus)/m_rho[j]
413  - rdt*(Y(x,k,j) - Y_prev(k,j));
414  diag[index(c_offset_Y + k, j)] = 1;
415  }
416 
417  //-----------------------------------------------
418  // energy equation
419  //
420  // \rho c_p dT/dt + \rho c_p u dT/dz
421  // = d(k dT/dz)/dz
422  // - sum_k(\omega_k h_k_ref)
423  // - sum_k(J_k c_p_k / M_k) dT/dz
424  //-----------------------------------------------
425  if (m_do_energy[j]) {
426  setGas(x,j);
427 
428  // heat release term
429  const vector_fp& h_RT = m_thermo->enthalpy_RT_ref();
430  const vector_fp& cp_R = m_thermo->cp_R_ref();
431  double sum = 0.0;
432  double sum2 = 0.0;
433  for (size_t k = 0; k < m_nsp; k++) {
434  double flxk = 0.5*(m_flux(k,j-1) + m_flux(k,j));
435  sum += wdot(k,j)*h_RT[k];
436  sum2 += flxk*cp_R[k]/m_wt[k];
437  }
438  sum *= GasConstant * T(x,j);
439  double dtdzj = dTdz(x,j);
440  sum2 *= GasConstant * dtdzj;
441 
442  rsd[index(c_offset_T, j)] = - m_cp[j]*rho_u(x,j)*dtdzj
443  - divHeatFlux(x,j) - sum - sum2;
444  rsd[index(c_offset_T, j)] /= (m_rho[j]*m_cp[j]);
445  rsd[index(c_offset_T, j)] -= rdt*(T(x,j) - T_prev(j));
446  rsd[index(c_offset_T, j)] -= (m_qdotRadiation[j] / (m_rho[j] * m_cp[j]));
447  diag[index(c_offset_T, j)] = 1;
448  } else {
449  // residual equations if the energy equation is disabled
450  rsd[index(c_offset_T, j)] = T(x,j) - T_fixed(j);
451  diag[index(c_offset_T, j)] = 0;
452  }
453 
454  rsd[index(c_offset_L, j)] = lambda(x,j) - lambda(x,j-1);
455  diag[index(c_offset_L, j)] = 0;
456  }
457  }
458 }

460 void StFlow::updateTransport(doublereal* x, size_t j0, size_t j1)
461 {
462  if (m_do_multicomponent) {
463  for (size_t j = j0; j < j1; j++) {
464  setGasAtMidpoint(x,j);
465  doublereal wtm = m_thermo->meanMolecularWeight();
466  doublereal rho = m_thermo->density();
467  m_visc[j] = (m_dovisc ? m_trans->viscosity() : 0.0);
468  m_trans->getMultiDiffCoeffs(m_nsp, &m_multidiff[mindex(0,0,j)]);
469 
470  // Use m_diff as storage for the factor outside the summation
471  for (size_t k = 0; k < m_nsp; k++) {
472  m_diff[k+j*m_nsp] = m_wt[k] * rho / (wtm*wtm);
473  }
474 
475  m_tcon[j] = m_trans->thermalConductivity();
476  if (m_do_soret) {
477  m_trans->getThermalDiffCoeffs(m_dthermal.ptrColumn(0) + j*m_nsp);
478  }
479  }
480  } else { // mixture averaged transport
481  for (size_t j = j0; j < j1; j++) {
482  setGasAtMidpoint(x,j);
483  m_visc[j] = (m_dovisc ? m_trans->viscosity() : 0.0);
484  m_trans->getMixDiffCoeffs(&m_diff[j*m_nsp]);
485  m_tcon[j] = m_trans->thermalConductivity();
486  }
487  }

507 void StFlow::updateDiffFluxes(const doublereal* x, size_t j0, size_t j1)
508 {
509  if (m_do_multicomponent) {
510  for (size_t j = j0; j < j1; j++) {
511  double dz = z(j+1) - z(j);
512  for (size_t k = 0; k < m_nsp; k++) {
513  doublereal sum = 0.0;
514  for (size_t m = 0; m < m_nsp; m++) {
515  sum += m_wt[m] * m_multidiff[mindex(k,m,j)] * (X(x,m,j+1)-X(x,m,j));
516  }
517  m_flux(k,j) = sum * m_diff[k+j*m_nsp] / dz;
518  }
519  }
520  } else {
521  for (size_t j = j0; j < j1; j++) {
522  double sum = 0.0;
523  double wtm = m_wtm[j];
524  double rho = density(j);
525  double dz = z(j+1) - z(j);
526  for (size_t k = 0; k < m_nsp; k++) {
527  m_flux(k,j) = m_wt[k]*(rho*m_diff[k+m_nsp*j]/wtm);
528  m_flux(k,j) *= (X(x,k,j) - X(x,k,j+1))/dz;
529  sum -= m_flux(k,j);
530  }
531  // correction flux to insure that \sum_k Y_k V_k = 0.
532  for (size_t k = 0; k < m_nsp; k++) {
533  m_flux(k,j) += sum*Y(x,k,j);
534  }
535  }
536  }
537 
538  if (m_do_soret) {
539  for (size_t m = j0; m < j1; m++) {
540  double gradlogT = 2.0 * (T(x,m+1) - T(x,m)) /
541  ((T(x,m+1) + T(x,m)) * (z(m+1) - z(m)));
542  for (size_t k = 0; k < m_nsp; k++) {
543  m_flux(k,m) -= m_dthermal(k,m)*gradlogT;
544  }
545  }
546  }

593 void StFlow::restore(const XML_Node& dom, doublereal* soln, int loglevel)
594 {
595  Domain1D::restore(dom, soln, loglevel);
596  vector<string> ignored;
597  size_t nsp = m_thermo->nSpecies();
598  vector_int did_species(nsp, 0);
599 
600  vector<XML_Node*> str = dom.getChildren("string");
601  for (size_t istr = 0; istr < str.size(); istr++) {
602  const XML_Node& nd = *str[istr];
603  writelog(nd["title"]+": "+nd.value()+"\n");
604  }
605 
606  double pp = getFloat(dom, "pressure", "pressure");
607  setPressure(pp);
608  vector<XML_Node*> d = dom.child("grid_data").getChildren("floatArray");
609  vector_fp x;
610  size_t np = 0;
611  bool readgrid = false, wrote_header = false;
612  for (size_t n = 0; n < d.size(); n++) {
613  const XML_Node& fa = *d[n];
614  string nm = fa["title"];
615  if (nm == "z") {
616  getFloatArray(fa,x,false);
617  np = x.size();
618  if (loglevel >= 2) {
619  writelog("Grid contains {} points.\n", np);
620  }
621  readgrid = true;
622  setupGrid(np, x.data());
623  }
624  }
625  if (!readgrid) {
626  throw CanteraError("StFlow::restore",
627  "domain contains no grid points.");
628  }
629 
630  debuglog("Importing datasets:\n", loglevel >= 2);
631  for (size_t n = 0; n < d.size(); n++) {
632  const XML_Node& fa = *d[n];
633  string nm = fa["title"];
634  getFloatArray(fa,x,false);
635  if (nm == "u") {
636  debuglog("axial velocity ", loglevel >= 2);
637  if (x.size() != np) {
638  throw CanteraError("StFlow::restore",
639  "axial velocity array size error");
640  }
641  for (size_t j = 0; j < np; j++) {
642  soln[index(c_offset_U,j)] = x[j];
643  }
644  } else if (nm == "z") {
645  ; // already read grid
646  } else if (nm == "V") {
647  debuglog("radial velocity ", loglevel >= 2);
648  if (x.size() != np) {
649  throw CanteraError("StFlow::restore",
650  "radial velocity array size error");
651  }
652  for (size_t j = 0; j < np; j++) {
653  soln[index(c_offset_V,j)] = x[j];
654  }
655  } else if (nm == "T") {
656  debuglog("temperature ", loglevel >= 2);
657  if (x.size() != np) {
658  throw CanteraError("StFlow::restore",
659  "temperature array size error");
660  }
661  for (size_t j = 0; j < np; j++) {
662  soln[index(c_offset_T,j)] = x[j];
663  }
664 
665  // For fixed-temperature simulations, use the imported temperature
666  // profile by default. If this is not desired, call
667  // setFixedTempProfile *after* restoring the solution.
668  vector_fp zz(np);
669  for (size_t jj = 0; jj < np; jj++) {
670  zz[jj] = (grid(jj) - zmin())/(zmax() - zmin());
671  }
672  setFixedTempProfile(zz, x);
673  } else if (nm == "L") {
674  debuglog("lambda ", loglevel >= 2);
675  if (x.size() != np) {
676  throw CanteraError("StFlow::restore",
677  "lambda arary size error");
678  }
679  for (size_t j = 0; j < np; j++) {
680  soln[index(c_offset_L,j)] = x[j];
681  }
682  } else if (m_thermo->speciesIndex(nm) != npos) {
683  debuglog(nm+" ", loglevel >= 2);
684  if (x.size() == np) {
685  size_t k = m_thermo->speciesIndex(nm);
686  did_species[k] = 1;
687  for (size_t j = 0; j < np; j++) {
688  soln[index(k+c_offset_Y,j)] = x[j];
689  }
690  }
691  } else {
692  ignored.push_back(nm);
693  }
694  }
695 
696  if (loglevel >=2 && !ignored.empty()) {
697  writelog("\n\n");
698  writelog("Ignoring datasets:\n");
699  size_t nn = ignored.size();
700  for (size_t n = 0; n < nn; n++) {
701  writelog(ignored[n]+" ");
702  }
703  }
704 
705  if (loglevel >= 1) {
706  for (size_t ks = 0; ks < nsp; ks++) {
707  if (did_species[ks] == 0) {
708  if (!wrote_header) {
709  writelog("Missing data for species:\n");
710  wrote_header = true;
711  }
712  writelog(m_thermo->speciesName(ks)+" ");
713  }
714  }
715  }
716 
717  if (dom.hasChild("energy_enabled")) {
718  getFloatArray(dom, x, false, "", "energy_enabled");
719  if (x.size() == nPoints()) {
720  for (size_t i = 0; i < x.size(); i++) {
721  m_do_energy[i] = (x[i] != 0);
722  }
723  } else if (!x.empty()) {
724  throw CanteraError("StFlow::restore", "energy_enabled is length {}"
725  "but should be length {}", x.size(), nPoints());
726  }
727  }
728 
729  if (dom.hasChild("species_enabled")) {
730  getFloatArray(dom, x, false, "", "species_enabled");
731  if (x.size() == m_nsp) {
732  for (size_t i = 0; i < x.size(); i++) {
733  m_do_species[i] = (x[i] != 0);
734  }
735  } else if (!x.empty()) {
736  // This may occur when restoring from a mechanism with a different
737  // number of species.
738  if (loglevel > 0) {
739  writelog("\nWarning: StFlow::restore: species_enabled is "
740  "length {} but should be length {}. Enabling all species "
741  "equations by default.", x.size(), m_nsp);
742  }
743  m_do_species.assign(m_nsp, true);
744  }
745  }
746 
747  if (dom.hasChild("refine_criteria")) {
748  XML_Node& ref = dom.child("refine_criteria");
749  refiner().setCriteria(getFloat(ref, "ratio"), getFloat(ref, "slope"),
750  getFloat(ref, "curve"), getFloat(ref, "prune"));
751  refiner().setGridMin(getFloat(ref, "grid_min"));

875 void AxiStagnFlow::evalRightBoundary(doublereal* x, doublereal* rsd,
876  integer* diag, doublereal rdt)
877 {
878  size_t j = m_points - 1;
879  // the boundary object connected to the right of this one may modify or
880  // replace these equations. The default boundary conditions are zero u, V,
881  // and T, and zero diffusive flux for all species.
882  rsd[index(c_offset_U,j)] = rho_u(x,j);
883  rsd[index(c_offset_V,j)] = V(x,j);
884  if (m_do_energy[j]) {
885  rsd[index(c_offset_T,j)] = T(x,j);
886  } else {
887  rsd[index(c_offset_T, j)] = T(x,j) - T_fixed(j);
888  }
889  rsd[index(c_offset_L, j)] = lambda(x,j) - lambda(x,j-1);
890  diag[index(c_offset_L, j)] = 0;
891  doublereal sum = 0.0;
892  for (size_t k = 0; k < m_nsp; k++) {
893  sum += Y(x,k,j);
894  rsd[index(k+c_offset_Y,j)] = m_flux(k,j-1) + rho_u(x,j)*Y(x,k,j);
895  }
896  rsd[index(c_offset_Y + rightExcessSpecies(), j)] = 1.0 - sum;
897  diag[index(c_offset_Y + rightExcessSpecies(), j)] = 0;
898 }

900 void AxiStagnFlow::evalContinuity(size_t j, doublereal* x, doublereal* rsd,
901  integer* diag, doublereal rdt)
902 {
903  //----------------------------------------------
904  // Continuity equation
905  //
906  // Note that this propagates the mass flow rate information to the left
907  // (j+1 -> j) from the value specified at the right boundary. The
908  // lambda information propagates in the opposite direction.
909  //
910  // d(\rho u)/dz + 2\rho V = 0
911  //------------------------------------------------
912  rsd[index(c_offset_U,j)] =
913  -(rho_u(x,j+1) - rho_u(x,j))/m_dz[j]
914  -(density(j+1)*V(x,j+1) + density(j)*V(x,j));
915 
916  //algebraic constraint
917  diag[index(c_offset_U, j)] = 0;
918 }

953 void FreeFlame::evalContinuity(size_t j, doublereal* x, doublereal* rsd,
954  integer* diag, doublereal rdt)
955 {
956  //----------------------------------------------
957  // Continuity equation
958  //
959  // d(\rho u)/dz + 2\rho V = 0
960  //----------------------------------------------
961  if (grid(j) > m_zfixed) {
962  rsd[index(c_offset_U,j)] =
963  - (rho_u(x,j) - rho_u(x,j-1))/m_dz[j-1]
964  - (density(j-1)*V(x,j-1) + density(j)*V(x,j));
965  } else if (grid(j) == m_zfixed) {
966  if (m_do_energy[j]) {
967  rsd[index(c_offset_U,j)] = (T(x,j) - m_tfixed);
968  } else {
969  rsd[index(c_offset_U,j)] = (rho_u(x,j)
970  - m_rho[0]*0.3);
971  }
972  } else if (grid(j) < m_zfixed) {
973  rsd[index(c_offset_U,j)] =
974  - (rho_u(x,j+1) - rho_u(x,j))/m_dz[j]
975  - (density(j+1)*V(x,j+1) + density(j)*V(x,j));
976  }
977  //algebraic constraint
978  diag[index(c_offset_U, j)] = 0;
979 }

981 void FreeFlame::_finalize(const doublereal* x)
982 {
983  StFlow::_finalize(x);
984  // If the domain contains the temperature fixed point, make sure that it
985  // is correctly set. This may be necessary when the grid has been modified
986  // externally.
987  if (m_tfixed != Undef) {
988  for (size_t j = 0; j < m_points; j++) {
989  if (z(j) == m_zfixed) {
990  return; // fixed point is already set correctly
991  }
992  }
993 
994  for (size_t j = 0; j < m_points - 1; j++) {
995  // Find where the temperature profile crosses the current
996  // fixed temperature.
997  if ((T(x, j) - m_tfixed) * (T(x, j+1) - m_tfixed) <= 0.0) {
998  m_tfixed = T(x, j+1);
999  m_zfixed = z(j+1);
1000  return;
1001  }
1002  }
1003  }
