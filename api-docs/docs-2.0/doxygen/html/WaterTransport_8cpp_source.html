96  AssertTrace(m_waterProps != 0);

98  } else {

100  if (!m_waterPDSS) {

102  "Expectation is that first species be water with a PDSS_Water object");

104  // Get a pointer to a changeable WaterPropsIAPWS object
105  m_sub = m_waterPDSS->getWater();
